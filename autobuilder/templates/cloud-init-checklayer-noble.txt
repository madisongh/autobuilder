    - sysctl -w kernel.apparmor_restrict_unprivileged_userns=0

    - [ sh, -c, "cd /var/lib/bwsetup/buildworker-setup-0.5.3; ./configure --prefix=/usr --disable-digsigserver && make && make install" ]
